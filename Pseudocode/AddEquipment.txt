START

IF countEq >= MAX_EQUIPMENT THEN

    DISPLAY "Storage Full"
    STOP

ENDIF

INPUT Equipment ID
INPUT Equipment Category
INPUT Power Consumption
INPUT Maximum Safe Temperature
INPUT Maintenance Interval

SET accumulatedHours = 0
SET hoursUsed = 0
SET temperature = 0
SET status = "NORMAL"

STORE equipment data into array

countEq = countEq + 1

DISPLAY "Equipment Added Successfully"

END
