START

SET totalEnergy = 0

FOR i = 0 TO countEq - 1

    energy = (power / 1000) * hoursUsed

    totalEnergy = totalEnergy + energy

END FOR

cost = totalEnergy * 0.50

IF totalEnergy > 8 THEN

    surcharge = cost * 0.20

ELSE

    surcharge = 0

ENDIF

finalCost = cost + surcharge

DISPLAY "Total Energy Consumption"
DISPLAY totalEnergy

DISPLAY "Electricity Cost"
DISPLAY cost

DISPLAY "Surcharge"
DISPLAY surcharge

DISPLAY "Final Cost"
DISPLAY finalCost

END
